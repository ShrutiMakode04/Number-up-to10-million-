1. TECHNICAL OVERVIEW

This document specifies the architecture, component design, state management,
data models, simulation logic, gamification implementation, audio pipeline,
and quality standards for the "Equal Groups — Introduction to Multiplication
and Division" interactive lesson module (Lesson 4.1) within Intellia SG's
Grade 1 Math program.

The module is a React 18 application (Vite + JSX), structured identically to
the reference repository https://github.com/dsamyak/numberbound, and styled
to match https://numberbound.vercel.app/. It will be embedded in WordPress at:

  https://intelliasg.com/courses/grade-1-math/lessons/equal-groups/

Audio narration uses ElevenLabs exclusively (no browser Web Speech API fallback),
mirroring the pipeline from the "Word Problems Using Addition Audio & Narration
Pipeline" document, adapted for this lesson's scripts.

4.3 Key Reducer Logic

// ANSWER_CORRECT dispatch
case ACTIONS.ANSWER_CORRECT: {
  const xpEarned = calcXP(state.attemptCount + 1, state.hintsUsed, state.streak);
  const newStreak = state.streak + 1;
  const worldIndex = Math.floor(state.currentQuestion / 10);
  const newWorldScore = (state.worldScores[worldIndex] || 0) + 1;
  const updatedWorldScores = [...state.worldScores];
  updatedWorldScores[worldIndex] = newWorldScore;
  return {
    ...state,
    xp: state.xp + xpEarned,
    streak: newStreak,
    maxStreak: Math.max(state.maxStreak, newStreak),
    worldScores: updatedWorldScores,
    totalStars: calcTotalStars(updatedWorldScores),
    hintsUsed: 0,
    attemptCount: 0,
  };
}

// ANSWER_INCORRECT dispatch
case ACTIONS.ANSWER_INCORRECT: {
  return {
    ...state,
    streak: 0,
    attemptCount: state.attemptCount + 1,
  };
}

5. QUESTION DATA MODEL

5.1 Question Schema

interface Question {
  id:           string;        // e.g. "Q1_003", "Q7_008"
  type:         QuestionType;  // One of 10 enum values (see below)
  world:        number;        // 0–9 (which world this belongs to)
  difficulty:   1 | 2 | 3;    // 1=easy(≤20), 2=medium(≤30), 3=hard(≤40)

  // Core math values
  numGroups:    number;        // Number of groups
  groupSize:    number;        // Size of each group
  total:        number;        // numGroups × groupSize
  missingSlot:  'numGroups' | 'groupSize' | 'total';

  // Rendering
  questionText: string;        // Full narrated question text (ElevenLabs reads this)
  visual:       VisualType;    // 'groupDiagram' | 'picture' | 'sentence' | 'trueFalse'
  objectEmoji?: string;        // E.g. '🍓', '🍪', '⭐', '🐟', '🎈'

  // MCQ
  options?:     (number|string)[];  // 4 MCQ options (always includes correctAnswer)

  // Hints
  hint1:        string;        // Shown after 1 wrong attempt
  hint2:        string;        // Shown after 2 wrong attempts (animation trigger)
  explanation:  string;        // Full text explanation after 3 fails (read aloud)

  // Word problems only
  characterName?:  string;
  containerName?:  string;     // 'plate', 'bag', 'box', 'basket'
  objectName?:     string;

  // True/False only
  isTrue?:      boolean;

  // Answer
  correctAnswer: number | string;
}

type QuestionType =
  | 'count_groups'         // Q1: Count total in equal groups
  | 'picture_group'        // Q2: Tap picture showing N groups of M
  | 'fill_total'           // Q3: N groups of M = ___
  | 'fill_group_size'      // Q4: N groups of ___ = total
  | 'fill_num_groups'      // Q5: ___ groups of M = total
  | 'word_problem_mult'    // Q6: multiplication-sense word problem
  | 'word_problem_div'     // Q7: fair-sharing word problem
  | 'true_false_group'     // Q8: Is "N groups of M = X" true or false?
  | 'spot_group_mcq'       // Q9: Which picture shows equal groups?
  | 'repeated_addition';   // Q10: Repeated addition link

type VisualType =
  | 'groupDiagram'         // SVG circles with emoji dots (EqualGroupDiagram)
  | 'picture'              // Static arrangement picture card
  | 'sentence'             // "___ groups of ___ = ___" with highlighted blank
  | 'trueFalse';           // Statement + True/False buttons

6. EQUAL GROUP DIAGRAM SVG COMPONENT

// EqualGroupDiagram.jsx — reusable SVG for N groups of M objects

const EqualGroupDiagram = ({
  numGroups,
  groupSize,
  missingSlot,
  objectEmoji = '⭐',
  animated = false,
  size = 'medium',  // 'small' | 'medium' | 'large'
}) => {

  // Layout: groups arranged in a row (up to 5) or 2 rows (6–9 groups)
  const circleRadius = size === 'large' ? 48 : size === 'medium' ? 38 : 28;
  const dotSize = size === 'large' ? 18 : size === 'medium' ? 14 : 10;
  const groupsPerRow = Math.min(numGroups, 5);
  const numRows = Math.ceil(numGroups / 5);
  const svgWidth = groupsPerRow * (circleRadius * 2 + 20) + 20;
  const svgHeight = numRows * (circleRadius * 2 + 40) + 40;

  return (
    <svg viewBox={`0 0 ${svgWidth} ${svgHeight}`}
         xmlns="http://www.w3.org/2000/svg"
         style={{ maxWidth: '100%', height: 'auto' }}>
      {Array(numGroups).fill(0).map((_, gi) => {
        const row = Math.floor(gi / 5);
        const col = gi % 5;
        const cx = col * (circleRadius * 2 + 20) + circleRadius + 10;
        const cy = row * (circleRadius * 2 + 40) + circleRadius + 20;

        return (
          <g key={gi}>
            {/* Group circle */}
            <circle
              cx={cx} cy={cy} r={circleRadius}
              fill={`hsl(${(gi * 45) % 360}, 70%, 90%)`}
              stroke={`hsl(${(gi * 45) % 360}, 70%, 60%)`}
              strokeWidth="2.5"
              strokeDasharray={missingSlot === 'groupSize' && gi === 0 ? '6,3' : '0'}
            />
            {/* Emoji dots arranged inside circle (up to groupSize) */}
            {Array(groupSize).fill(0).map((_, di) => {
              const angle = (di / groupSize) * 2 * Math.PI - Math.PI / 2;
              const r = groupSize === 1 ? 0 : circleRadius * 0.55;
              const dx = cx + r * Math.cos(angle);
              const dy = cy + r * Math.sin(angle);
              return (
                <text key={di} x={dx} y={dy + dotSize / 3}
                      textAnchor="middle" fontSize={dotSize}>
                  {objectEmoji}
                </text>
              );
            })}
            {/* Group count label below circle */}
            <text x={cx} y={cy + circleRadius + 16}
                  textAnchor="middle" fontSize="13"
                  fill="#555" fontWeight="500">
              {groupSize}
            </text>
          </g>
        );
      })}
      {/* Total label underneath */}
      <text
        x={svgWidth / 2} y={svgHeight - 6}
        textAnchor="middle" fontSize="15"
        fill="#333" fontWeight="bold">
        {`${numGroups} groups of ${groupSize} = ${missingSlot === 'total' ? '?' : numGroups * groupSize}`}
      </text>
    </svg>
  );
};

Animation variants:
  animated=true → CSS 'dotCountUp' keyframe: dots appear one-by-one with 80ms delay per dot
  shake variant  → CSS 'shake' keyframe applied to <svg> wrapper on wrong answer
  bounce variant → CSS 'bounceIn' keyframe applied to <svg> wrapper on correct answer

7. SIMULATION STATION COMPONENT SPECS

7.1 GroupingStation.jsx — Station A (Concrete)

State:
  const [groupConfig, setGroupConfig] = useState(getStationARound(state.simRound));
  // groupConfig: { numGroups: 3, groupSize: 4, objectEmoji: '🍪' }
  const [placed, setPlaced] = useState(
    Array(groupConfig.numGroups).fill(0)  // Count of objects in each circle
  );
  const [objectPool, setObjectPool] = useState(
    Array(groupConfig.numGroups * groupConfig.groupSize).fill(groupConfig.objectEmoji)
  );

Interaction (Drag):
  - ObjectPool renders draggable <span> elements for each object
  - GroupCircle renders a droppable <div> zone
  - onDrop: placed[circleIndex] += 1; remove object from pool
  - All circles show counter badge: "2 / 4"

Interaction (Tap fallback):
  - Tap object in pool → object becomes "selected" (glows)
  - Tap circle → object moves into circle

Completion Check:
  - placed.every(count => count === groupConfig.groupSize) → all equal
  - Submit button appears only when total objects placed === total
  - On submit correct: mascot celebrates, ElevenLabs plays celebration audio
  - On submit unequal: shake + narration "One group has more! Let's try again."

Station A Rounds (4 rounds, randomised order):
  { numGroups: 2, groupSize: 3, objectEmoji: '🍪' }
  { numGroups: 3, groupSize: 4, objectEmoji: '⭐' }
  { numGroups: 4, groupSize: 2, objectEmoji: '🐟' }
  { numGroups: 5, groupSize: 3, objectEmoji: '🎈' }

7.2 SpotEqualStation.jsx — Station B (Pictorial)

State:
  const [arrangements, setArrangements] = useState(generateArrangementCards(round));
  const [selected, setSelected] = useState([]);   // Indices of tapped cards
  const [submitted, setSubmitted] = useState(false);

Card Generation (generateArrangementCards):
  - Creates 4 arrangement cards: 2–3 have equal groups, 1–2 are unequal
  - Each card shows numGroups circles, each with randomised dot counts
  - Equal card: all groups have same count (e.g. 3, 3, 3)
  - Unequal card: one group differs by ±1 (e.g. 3, 3, 4) — plausible distractor
  - Objects rendered as emoji text inside group circles (no numbers shown)

Interaction:
  - Tap a card → border highlights (selected state)
  - Multi-select allowed (student picks all equal ones)
  - "Check" button submits selection
  - On submit: correct cards glow green, wrong cards glow red (1.5s)
  - Then advance to next round or next station

Rounds (3 rounds per station):
  Round 1: 2-group arrangements (easy visual split)
  Round 2: 3-group arrangements (medium — more counting needed)
  Round 3: 4-group arrangements (hard — attention required for small differences)

State:

    {missingSlot === 'groupSize'
      ? <BlankInput value={inputValue} />
      : <span class="given-value">{groupSize}</span>
    }
    <span class="equals">=</span>
    {missingSlot === 'total'
      ? <BlankInput value={inputValue} />
      : <span class="given-value">{total}</span>
    }
  </div>
  <NumberPad max={40} value={inputValue} onChange={setInputValue} onSubmit={handleSubmit} />
  <button onClick={() => setShowDiagram(!showDiagram)}>Show me the groups 🔢</button>
  {showDiagram && <EqualGroupDiagram numGroups={numGroups} groupSize={groupSize}
                    missingSlot={missingSlot} animated />}

Variants (rotated across 3 rounds):
  Round 1: Find total      → 3 groups of 4 = ___
  Round 2: Find group size → 2 groups of ___ = 10
  Round 3: Find num groups → ___ groups of 5 = 15

ElevenLabs reads the full sentence aloud when displayed:
  "Three groups of four equals what? Type the answer!"

8.4 Frontend Audio Engine (src/hooks/useAudio.js)

// Step 1: Check audioMap for pre-generated static asset
// Step 2: If not found + API key present → fetch from ElevenLabs dynamically
// Step 3: Cache dynamic result in elevenLabsCache (in-memory Map)
// Step 4: Play via HTML5 Audio API (new Audio(url))
// Step 5: While segment i plays → preload segment i+1 (eager preload)

const elevenLabsCache = new Map();  // In-memory; cleared on page refresh

export async function getAudioUrl(text, style = 'statement', apiKey) {
  // 1. Static map check (fastest path)
  if (audioMap[text]) return audioMap[text];

  // 2. Memory cache check
  const cacheKey = `${text}::${style}`;
  if (elevenLabsCache.has(cacheKey)) return elevenLabsCache.get(cacheKey);

  // 3. Dynamic generation (requires API key)
  if (!apiKey) return null;  // Silent skip — no fallback

  const styleSettings = STYLE_SETTINGS[style] ?? STYLE_SETTINGS.statement;
  const response = await fetch(
    `https://api.elevenlabs.io/v1/text-to-speech/Xb7hH8MSUJpSbSDYk0k2`,
    {
      method: 'POST',
      headers: {
        'xi-api-key': apiKey,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        text,
        model_id: 'eleven_multilingual_v2',
        voice_settings: styleSettings,
      }),
    }
  );
  const blob = await response.blob();
  const url = URL.createObjectURL(blob);
  elevenLabsCache.set(cacheKey, url);
  return url;
}

export async function narrate(segments, apiKey, onSegmentStart) {
  for (let i = 0; i < segments.length; i++) {
    const { text, style } = segments[i];
    const url = await getAudioUrl(text, style, apiKey);
    if (!url) continue;  // Silent skip if no audio available

    // Eager preload next segment
    if (i + 1 < segments.length) {
      getAudioUrl(segments[i + 1].text, segments[i + 1].style, apiKey);
    }

    if (onSegmentStart) onSegmentStart(i);
    await playAudio(url);  // Resolves on 'ended' event
  }
}

async function playAudio(url) {
  return new Promise((resolve) => {
    const audio = new Audio(url);
    audio.onended = resolve;
    audio.onerror = resolve;  // Silent fail — never block UX
    audio.play().catch(resolve);
  });
}

9. RANDOMISATION ENGINE

9.1 Fisher-Yates Shuffle (utils/shuffle.js)

export function shuffleArray(array) {
  const arr = [...array];
  for (let i = arr.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [arr[i], arr[j]] = [arr[j], arr[i]];
  }
  return arr;
}

export function generateSessionQuestions(bank) {
  const byType = {};
  bank.forEach(q => {
    if (!byType[q.type]) byType[q.type] = [];
    byType[q.type].push(q);
  });
  // Pick 10 from each type (shuffled), then shuffle the combined 100
  const selected = Object.values(byType)
    .flatMap(qs => shuffleArray(qs).slice(0, 10));
  return shuffleArray(selected);
}

9.2 MCQ Distractor Generation (utils/scoring.js)

export function generateDistractors(correct, min = 0, max = 40, count = 3) {
  const distractors = new Set();
  // Strategy: offsets of ±1, ±2, ±3 — plausible wrong group counts
  const offsets = [-3, -2, -1, 1, 2, 3];
  shuffleArray(offsets).forEach(offset => {
    const d = correct + offset;
    if (d >= min && d <= max && d !== correct && distractors.size < count)
      distractors.add(d);
  });
  // Ensure always 4 options
  while (distractors.size < count) {
    const d = correct + (distractors.size + 1);
    if (d <= max && d !== correct) distractors.add(d);
  }
  return shuffleArray([correct, ...distractors]);
}

export function calcXP(attemptNumber, hintsUsed, streak) {
  const base = attemptNumber === 1 ? 10 : hintsUsed > 0 ? 5 : 7;
  const streakBonus = streak >= 5 ? 5 : 0;
  return base + streakBonus;
}

10.2 Star Rating (per world of 10 questions)

export function calcStars(correct, total = 10) {
  if (correct >= 9) return 3;    // Gold: ≥90%
  if (correct >= 7) return 2;    // Silver: ≥70%
  if (correct >= 5) return 1;    // Bronze: ≥50% (world unlock gate)
  return 0;                       // Try again
}

export function canUnlockWorld(worldScore) {
  return worldScore !== null && worldScore >= 5;
}

export function calcTotalStars(worldScores) {
  return worldScores.reduce((sum, ws) => sum + (ws !== null ? calcStars(ws) : 0), 0);
}
